Runtime events
runOnStartup(async runtime =>
{
  runtime.addEventListener("beforeprojectstart", () => OnBeforeProjectStart(runtime));
  runtime.addEventListener("afterprojectstart", () => OnAfterProjectStart(runtime));
});

function OnBeforeProjectStart(runtime)
{
  runtime.addEventListener("tick", () => OnTick(runtime));
  runtime.addEventListener("beforelayoutstart", () => OnBeforeLayoutStart(runtime));
  runtime.addEventListener("afterlayoutstart", () => OnAfterLayoutStart(runtime));

  runtime.addEventListener("keydown", () => OnKeyDown(runtime));
  runtime.addEventListener("keyup", () => OnKeyUp(runtime));

  runtime.addEventListener("mousedown", () => OnMouseDown(runtime));
  runtime.addEventListener("mousemove", () => OnMouseMove(runtime));
  runtime.addEventListener("mouseup", () => OnMouseUp(runtime));
  runtime.addEventListener("dblclick", () => OnMouseDoubleClick(runtime));
  runtime.addEventListener("wheel", () => OnMouseWheel(runtime));

  runtime.addEventListener("pointerdown", () => OnPointDown(runtime));
  runtime.addEventListener("pointermove", () => OnPointMove(runtime));
  runtime.addEventListener("pointerup", () => OnPointUp(runtime));
  runtime.addEventListener("pointercancel", () => OnPointCancel(runtime));

  runtime.addEventListener("deviceorientation", () => OnDeviceOrientation(runtime));
  runtime.addEventListener("devicemotion", () => OnDevicemotion(runtime));

  runtime.addEventListener("save", () => OnSave(runtime));
  runtime.addEventListener("load", () => OnLoad(runtime));

  runtime.addEventListener("instancecreate", () => OnInstanceCreate(runtime));
  runtime.addEventListener("instancedestroy", () => OnInstanceDestroy(runtime));
}
